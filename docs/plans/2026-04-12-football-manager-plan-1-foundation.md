# Football Manager — Plan 1: Foundation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Set up the project scaffold, TypeScript types, SQLite database with full schema, a seeded RNG, and a procedural data generation script that populates 96 teams with ~2400 players across 5 European leagues.

**Architecture:** Expo + TypeScript project with expo-sqlite for persistence. Engine logic lives in `src/engine/` as pure TypeScript (no React imports). Database layer in `src/database/` with typed queries. Seed data generated by a standalone script in `scripts/` and bundled as JSON assets imported on first boot.

**Tech Stack:** Expo SDK 54, React Native 0.81+, TypeScript 5.9+, expo-sqlite, zustand, jest + ts-jest

Create `src/engine/rng.ts` using the mulberry32 algorithm:

```ts
/**
 * Seeded pseudo-random number generator using mulberry32.
 * Deterministic: same seed always produces same sequence.
 */
export class SeededRng {
  private state: number;

  constructor(seed: number) {
    this.state = seed | 0;
  }

  /** Returns a float in [0, 1) */
  next(): number {
    this.state = (this.state + 0x6d2b79f5) | 0;
    let t = Math.imul(this.state ^ (this.state >>> 15), 1 | this.state);
    t = (t + Math.imul(t ^ (t >>> 7), 61 | t)) ^ t;
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  }

  /** Returns an integer in [min, max] (inclusive) */
  nextInt(min: number, max: number): number {
    return Math.floor(this.next() * (max - min + 1)) + min;
  }

  /** Returns a float in [min, max) */
  nextFloat(min: number, max: number): number {
    return this.next() * (max - min) + min;
  }

  /** Picks a random element from an array */
  pick<T>(array: readonly T[]): T {
    return array[Math.floor(this.next() * array.length)];
  }

  /** Shuffles an array in place (Fisher-Yates) and returns it */
  shuffle<T>(array: T[]): T[] {
    for (let i = array.length - 1; i > 0; i--) {
      const j = Math.floor(this.next() * (i + 1));
      [array[i], array[j]] = [array[j], array[i]];
    }
    return array;
  }

  /** Picks an element with weighted probability */
  weightedPick<T>(items: readonly T[], weights: readonly number[]): T {
    const total = weights.reduce((sum, w) => sum + w, 0);
    let roll = this.next() * total;
    for (let i = 0; i < items.length; i++) {
      roll -= weights[i];
      if (roll <= 0) return items[i];
    }
    return items[items.length - 1];
  }
}
```

 * Returns a number 1-99 representing how good the player is at the given position.
 */
export function calculateOverall(attributes: PlayerAttributes, position: Position): number {

  let totalWeight = 0;
  let weightedSum = 0;

  for (const key of ATTRIBUTE_KEYS) {
    const w = weights[key];
    totalWeight += w;
    weightedSum += attributes[key] * w;
  }

  if (totalWeight === 0) return 1;

  const raw = weightedSum / totalWeight;
  return Math.round(Math.max(1, Math.min(99, raw)));
}

import * as SQLite from 'expo-sqlite';
import { createAllTables } from './schema';

const DB_NAME = 'football-manager.db';

let db: SQLite.SQLiteDatabase | null = null;

/**
 * Opens (or returns cached) database connection and ensures schema exists.
 */
export async function getDatabase(): Promise<SQLite.SQLiteDatabase> {
  if (db) return db;
  db = await SQLite.openDatabaseAsync(DB_NAME);
  await db.execAsync('PRAGMA journal_mode = WAL;');
  await db.execAsync('PRAGMA foreign_keys = ON;');
  createAllTables({ exec: (sql: string) => db!.execSync(sql) });
  return db;
}

/**
 * Closes the database connection.
 */
export async function closeDatabase(): Promise<void> {
  if (db) {
    await db.closeAsync();
    db = null;
  }
}

  'CDM', 'CDM', 'CM', 'CM', 'CM', 'CAM', 'LM', 'RM',
  'LW', 'RW', 'ST', 'ST', 'ST',
];

// Secondary positions by primary
const SECONDARY_POSITION_MAP: Partial<Record<Position, Position[]>> = {
  CB: ['CDM'],
  LB: ['LM', 'LW'],
  RB: ['RM', 'RW'],
  CDM: ['CB', 'CM'],
  CM: ['CDM', 'CAM'],
  CAM: ['CM', 'LW', 'RW'],
  LM: ['LW', 'LB'],
  RM: ['RW', 'RB'],
  LW: ['LM', 'ST'],
  RW: ['RM', 'ST'],
  ST: ['CAM', 'LW', 'RW'],
};

const STAFF_ROLES: StaffRole[] = ['scout', 'physio', 'assistant', 'youth_coach', 'fitness_coach'];

const FORMATIONS: Formation[] = ['4-4-2', '4-3-3', '4-2-3-1', '3-5-2', '4-5-1'];

function generateName(rng: SeededRng, nationality: string): string {
  const firsts = FIRST_NAMES[nationality] || FIRST_NAMES.English;
  const lasts = LAST_NAMES[nationality] || LAST_NAMES.English;
  return `${rng.pick(firsts)} ${rng.pick(lasts)}`;
}

function pickNationality(rng: SeededRng, countryCode: string): string {
  const nat = NATIONALITIES_BY_COUNTRY[countryCode];
  if (!nat) return 'English';
  // 65% local, 35% foreign
  if (rng.next() < 0.65) return nat.primary;
  return rng.pick(nat.secondary);
}

function generateAttributes(rng: SeededRng, targetOverall: number, position: Position): SeedPlayerAttributes {
  // Generate attributes around target with variance
  const variance = 12;
  const gen = () => {
    const val = targetOverall + rng.nextInt(-variance, variance);
    return Math.max(1, Math.min(99, val));
  };

  const attrs: SeedPlayerAttributes = {
    playerId: 0, // set later
    finishing: gen(), passing: gen(), crossing: gen(),
    dribbling: gen(), heading: gen(), longShots: gen(), freeKicks: gen(),
    vision: gen(), composure: gen(), decisions: gen(),
    positioning: gen(), aggression: gen(), leadership: gen(),
    pace: gen(), stamina: gen(), strength: gen(),
    agility: gen(), jumping: gen(),
  };

  // Boost position-relevant attributes
  const boost = (key: keyof SeedPlayerAttributes, amount: number) => {
    if (key === 'playerId') return;
    attrs[key] = Math.min(99, (attrs[key] as number) + amount);
  };

  switch (position) {
    case 'GK':
      boost('composure', 8); boost('positioning', 10); boost('agility', 6); boost('jumping', 6);
      attrs.finishing = Math.max(1, Math.min(30, attrs.finishing - 20));
      break;
    case 'CB':
      boost('heading', 8); boost('strength', 8); boost('positioning', 6); boost('aggression', 5);
      break;
    case 'LB': case 'RB':
      boost('pace', 8); boost('stamina', 6); boost('crossing', 6);
      break;
    case 'CDM':
      boost('passing', 5); boost('positioning', 6); boost('strength', 6); boost('stamina', 5); boost('aggression', 4);
      break;
    case 'CM':
      boost('passing', 8); boost('vision', 6); boost('stamina', 6);
      break;
    case 'CAM':
      boost('passing', 6); boost('vision', 8); boost('dribbling', 6); boost('composure', 5);
      break;
    case 'LM': case 'RM':
      boost('pace', 6); boost('crossing', 6); boost('dribbling', 5); boost('stamina', 5);
      break;
    case 'LW': case 'RW':
      boost('pace', 8); boost('dribbling', 8); boost('agility', 5);
      break;
    case 'ST':
      boost('finishing', 10); boost('composure', 6); boost('positioning', 6);
      break;
  }

  return attrs;
}

function targetOverallForClub(rng: SeededRng, reputation: number, isStarPlayer: boolean): number {
  if (reputation >= 85) {
    return isStarPlayer ? rng.nextInt(82, 90) : rng.nextInt(72, 80);
  } else if (reputation >= 60) {
    return isStarPlayer ? rng.nextInt(76, 83) : rng.nextInt(65, 74);
  } else {
    return isStarPlayer ? rng.nextInt(68, 75) : rng.nextInt(55, 67);
  }
}

function generateAge(rng: SeededRng): number {
  // Weighted distribution: more players 22-30
  const roll = rng.next();
  if (roll < 0.12) return rng.nextInt(16, 19);      // 12% young
  if (roll < 0.35) return rng.nextInt(20, 23);       // 23% developing
  if (roll < 0.70) return rng.nextInt(24, 28);       // 35% prime
  if (roll < 0.90) return rng.nextInt(29, 32);       // 20% experienced
  return rng.nextInt(33, 36);                          // 10% veteran
}

function calculateMarketValue(age: number, overall: number, potential: number): number {
  let base = Math.pow(overall / 10, 3) * 100000; // ~34M for overall 70
  // Age multiplier
  if (age <= 21) base *= 1.5;
  else if (age <= 25) base *= 1.3;
  else if (age <= 28) base *= 1.1;
  else if (age <= 30) base *= 0.8;
  else if (age <= 33) base *= 0.5;
  else base *= 0.3;
  // Potential bonus
  const potentialGap = Math.max(0, potential - overall);
  base *= 1 + potentialGap * 0.03;
  return Math.round(base / 10000) * 10000; // round to 10k
}

function calculateWage(overall: number, clubReputation: number): number {
  const base = Math.pow(overall / 10, 2.5) * 500;
  const clubMultiplier = 0.8 + (clubReputation / 100) * 0.4;
  return Math.round(base * clubMultiplier / 100) * 100; // round to 100
}

export function generateSeedData(seed: number): SeedData {
  const rng = new SeededRng(seed);

  const countries: SeedCountry[] = [];
  const leagues: SeedLeague[] = [];
  const clubs: SeedClub[] = [];
  const players: SeedPlayer[] = [];
  const playerAttributes: SeedPlayerAttributes[] = [];
  const staff: SeedStaff[] = [];
  const tactics: SeedTactic[] = [];

  let countryId = 1;
  let leagueId = 1;
  let clubId = 1;
  let playerId = 1;
  let staffId = 1;
  let tacticId = 1;

  for (const leagueDef of LEAGUES) {
    // Country
    const country: SeedCountry = {
      id: countryId,
      name: leagueDef.country,
      code: leagueDef.countryCode,
      continent: 'Europe',
    };
    countries.push(country);

    // League
    const league: SeedLeague = {
      id: leagueId,
      name: leagueDef.name,
      countryId: countryId,
      divisionLevel: 1,
      numTeams: leagueDef.teams.length,
      promotionSpots: 0,
      relegationSpots: 3,
    };
    leagues.push(league);

    // Teams
    for (const teamDef of leagueDef.teams) {
      const facilityLevel = teamDef.reputation >= 85 ? rng.nextInt(4, 5) :
        teamDef.reputation >= 60 ? rng.nextInt(2, 4) : rng.nextInt(1, 3);

      const club: SeedClub = {
        id: clubId,
        name: teamDef.name,
        shortName: teamDef.shortName,
        countryId: countryId,
        leagueId: leagueId,
        reputation: teamDef.reputation,
        budget: teamDef.reputation * teamDef.reputation * 1000,
        wageBudget: teamDef.reputation * teamDef.reputation * 200,
        stadiumName: teamDef.stadiumName,
        stadiumCapacity: teamDef.stadiumCapacity,
        trainingFacilities: facilityLevel,
        youthAcademy: Math.max(1, facilityLevel + rng.nextInt(-1, 1)),
        medicalDepartment: Math.max(1, facilityLevel + rng.nextInt(-1, 0)),
        primaryColor: teamDef.primaryColor,
        secondaryColor: teamDef.secondaryColor,
      };
      clubs.push(club);

      // Squad
      const squadSize = rng.nextInt(23, 26);
      const positions = rng.shuffle([...SQUAD_TEMPLATE]);
      const squadPositions = positions.slice(0, squadSize);
      const starPlayerCount = teamDef.reputation >= 85 ? rng.nextInt(3, 4) :
        teamDef.reputation >= 60 ? rng.nextInt(1, 2) : rng.nextInt(0, 1);

      for (let i = 0; i < squadPositions.length; i++) {
        const pos = squadPositions[i];
        const isStar = i < starPlayerCount;
        const nationality = pickNationality(rng, leagueDef.countryCode);
        const age = generateAge(rng);
        const targetOvr = targetOverallForClub(rng, teamDef.reputation, isStar);

        // Young players may not have reached their target yet
        const ageReduction = age <= 19 ? rng.nextInt(5, 12) : age <= 22 ? rng.nextInt(2, 6) : 0;
        const currentOverall = Math.max(35, targetOvr - ageReduction);

        const basePotential = Math.min(99, currentOverall + rng.nextInt(2, 20));
        const effectivePotential = basePotential;

        const secondaryOptions = SECONDARY_POSITION_MAP[pos];
        const secondaryPosition = secondaryOptions && rng.next() < 0.4
          ? rng.pick(secondaryOptions) : null;

        const player: SeedPlayer = {
          id: playerId,
          name: generateName(rng, nationality),
          nationality,
          age,
          position: pos,
          secondaryPosition,
          clubId: clubId,
          wage: calculateWage(currentOverall, teamDef.reputation),
          contractEnd: rng.nextInt(1, 4), // 1-4 seasons from start
          marketValue: calculateMarketValue(age, currentOverall, basePotential),
          basePotential,
          effectivePotential,
          morale: rng.nextInt(55, 80),
          fitness: rng.nextInt(85, 100),
          injuryWeeksLeft: 0,
          isFreeAgent: false,
        };
        players.push(player);

        const attrs = generateAttributes(rng, currentOverall, pos);
        attrs.playerId = playerId;
        playerAttributes.push(attrs);

        playerId++;
      }

      // Staff (3-5 per club)
      const staffCount = rng.nextInt(3, 5);
      const availableRoles = rng.shuffle([...STAFF_ROLES]);
      for (let i = 0; i < staffCount; i++) {
        const role = availableRoles[i % availableRoles.length];
        const nationality = pickNationality(rng, leagueDef.countryCode);
        const ability = teamDef.reputation >= 85 ? rng.nextInt(12, 18) :
          teamDef.reputation >= 60 ? rng.nextInt(8, 15) : rng.nextInt(4, 12);

        staff.push({
          id: staffId,
          name: generateName(rng, nationality),
          role,
          clubId: clubId,
          ability,
          wage: ability * rng.nextInt(400, 800),
          contractEnd: rng.nextInt(1, 3),
        });
        staffId++;
      }

      // Default tactic
      tactics.push({
        id: tacticId,
        clubId: clubId,
        name: 'Default',
        isActive: true,
        formation: rng.pick(FORMATIONS),
        mentality: 'balanced',
        pressing: 'medium',
        passingStyle: 'mixed',
        tempo: 'normal',
        width: 'normal',
      });
      tacticId++;

      clubId++;
    }

    countryId++;
    leagueId++;
  }

  return { countries, leagues, clubs, players, playerAttributes, staff, tactics };
}

type PlayerRow = Record<string, unknown>;

function rowToPlayer(row: PlayerRow): Player {
  return {
    id: row.id as number,
    name: row.name as string,
    nationality: row.nationality as string,
    age: row.age as number,
    position: row.position as Position,
    secondaryPosition: (row.secondary_position as Position) || null,
    clubId: row.club_id as number,

    fitness: row.fitness as number,
    injuryWeeksLeft: row.injury_weeks_left as number,
    isFreeAgent: (row.is_free_agent as number) === 1,
  };
}

function rowToAttributes(row: PlayerRow): PlayerAttributes {
  return {
    finishing: row.finishing as number,
    passing: row.passing as number,
    crossing: row.crossing as number,
    dribbling: row.dribbling as number,
    heading: row.heading as number,
    longShots: row.long_shots as number,
    freeKicks: row.free_kicks as number,
    vision: row.vision as number,
    composure: row.composure as number,
    decisions: row.decisions as number,
    positioning: row.positioning as number,
    aggression: row.aggression as number,
    leadership: row.leadership as number,
    pace: row.pace as number,
    stamina: row.stamina as number,
    strength: row.strength as number,
    agility: row.agility as number,
    jumping: row.jumping as number,
  };
}

export function getPlayersByClub(db: DbHandle, clubId: number): Player[] {
  const rows = db.prepare('SELECT * FROM players WHERE club_id = ? ORDER BY position, name').all(clubId) as PlayerRow[];
  return rows.map(rowToPlayer);
}

export function getPlayerById(db: DbHandle, playerId: number): (Player & { attributes: PlayerAttributes }) | null {
  const row = db.prepare('SELECT * FROM players WHERE id = ?').get(playerId) as PlayerRow | undefined;
  if (!row) return null;
  const attrRow = db.prepare('SELECT * FROM player_attributes WHERE player_id = ?').get(playerId) as PlayerRow;
  return { ...rowToPlayer(row), attributes: rowToAttributes(attrRow) };
}

export interface SearchPlayersFilters {
  position?: Position;
  minAge?: number;
  maxAge?: number;
  clubId?: number;
  minOverall?: number;
  maxWage?: number;
}

export function searchPlayers(db: DbHandle, filters: SearchPlayersFilters): Player[] {
  const conditions: string[] = [];
  const params: unknown[] = [];

  if (filters.position) { conditions.push('position = ?'); params.push(filters.position); }
  if (filters.minAge !== undefined) { conditions.push('age >= ?'); params.push(filters.minAge); }
  if (filters.maxAge !== undefined) { conditions.push('age <= ?'); params.push(filters.maxAge); }
  if (filters.clubId !== undefined) { conditions.push('club_id = ?'); params.push(filters.clubId); }
  if (filters.maxWage !== undefined) { conditions.push('wage <= ?'); params.push(filters.maxWage); }

  const where = conditions.length > 0 ? `WHERE ${conditions.join(' AND ')}` : '';
  const rows = db.prepare(`SELECT * FROM players ${where} ORDER BY market_value DESC`).all(...params) as PlayerRow[];
  return rows.map(rowToPlayer);
}

export function updatePlayerMorale(db: DbHandle, playerId: number, morale: number): void {
  db.prepare('UPDATE players SET morale = ? WHERE id = ?').run(morale, playerId);
}

export function getFreeAgents(db: DbHandle): Player[] {
  const rows = db.prepare('SELECT * FROM players WHERE is_free_agent = 1 ORDER BY market_value DESC').all() as PlayerRow[];
  return rows.map(rowToPlayer);
}

  const rows = db.prepare('SELECT type, SUM(amount) as total FROM club_finances WHERE club_id = ? AND season = ? GROUP BY CASE WHEN amount >= 0 THEN \'income\' ELSE \'expense\' END').all(clubId, season) as Row[];
  let income = 0;
  let expenses = 0;
  for (const r of rows) {
    const total = r.total as number;
    if (total >= 0) income += total;

  }
  return { income, expenses, net: income - expenses };

interface DbHandle {
  prepare(sql: string): {
    run(...params: unknown[]): void;
  };
  exec(sql: string): void;
}

/**
 * Seeds the database with generated data inside a single transaction.
 * Throws on any error, rolling back all changes.
 */
export function seedDatabase(db: DbHandle, data: SeedData): void {
  db.exec('BEGIN TRANSACTION');

  try {
    const insertCountry = db.prepare('INSERT INTO countries (id, name, code, continent) VALUES (?, ?, ?, ?)');
    for (const c of data.countries) insertCountry.run(c.id, c.name, c.code, c.continent);

    const insertLeague = db.prepare('INSERT INTO leagues (id, name, country_id, division_level, num_teams, promotion_spots, relegation_spots) VALUES (?, ?, ?, ?, ?, ?, ?)');
    for (const l of data.leagues) insertLeague.run(l.id, l.name, l.countryId, l.divisionLevel, l.numTeams, l.promotionSpots, l.relegationSpots);

    const insertClub = db.prepare('INSERT INTO clubs (id, name, short_name, country_id, league_id, reputation, budget, wage_budget, stadium_name, stadium_capacity, training_facilities, youth_academy, medical_department, primary_color, secondary_color) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)');
    for (const c of data.clubs) insertClub.run(c.id, c.name, c.shortName, c.countryId, c.leagueId, c.reputation, c.budget, c.wageBudget, c.stadiumName, c.stadiumCapacity, c.trainingFacilities, c.youthAcademy, c.medicalDepartment, c.primaryColor, c.secondaryColor);

    const insertPlayer = db.prepare('INSERT INTO players (id, name, nationality, age, position, secondary_position, club_id, wage, contract_end, market_value, base_potential, effective_potential, morale, fitness, injury_weeks_left, is_free_agent) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)');
    for (const p of data.players) insertPlayer.run(p.id, p.name, p.nationality, p.age, p.position, p.secondaryPosition, p.clubId, p.wage, p.contractEnd, p.marketValue, p.basePotential, p.effectivePotential, p.morale, p.fitness, p.injuryWeeksLeft, p.isFreeAgent ? 1 : 0);

    const insertAttrs = db.prepare('INSERT INTO player_attributes (player_id, finishing, passing, crossing, dribbling, heading, long_shots, free_kicks, vision, composure, decisions, positioning, aggression, leadership, pace, stamina, strength, agility, jumping) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)');
